mse-eme-capabilities.js

This module can be used to check the browser capabilities regarding MSE and EME support.

Quick Start

Install Dependencies

  1. install nodejs
  2. install gulp
npm install -g gulp

Build / Run

npm run build

Documentation

The API documentation can be generated using following gulp command:

npm run doc

Getting Started

Include module "mseemecapabilities",

In order to simply check if the HTML5 MSE/EME based player hasplayer.js can be used to play contents on the current browser, you can use the following method:

(function(){
    // Check if clear contents are supported
    hasplayerCanPlay(false).then(function (result) {
        console.log('canPlay: ' + result.canPlay);
        if (error) {
            console.log('Error: ' + result.error.code + ' - ' + result.error.message);
        }
    });

    // Check if encrypted contents are supported
    hasplayerCanPlay(true).then(function (result) {
        console.log('canPlay: ' + result.canPlay);
        if (error) {
            console.log('Error: ' + result.error.code + ' - ' + result.error.message);
        }
    });

    // Check if encrypted HLS contents are supported
    hasplayerCanPlay(true, 'HLS').then(function (result) {
        console.log('canPlay: ' + result.canPlay);
        if (error) {
            console.log('Error: ' + result.error.code + ' - ' + result.error.message);
        }
    });
})();
